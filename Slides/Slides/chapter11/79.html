const TAX_RATE = 0.088; // 기타소득세 8.8% function calculateSettlement(revenue: number, type: SellerType) { const platformFee = revenue * 0.30; // 플랫폼 수수료 30% const taxableIncome = revenue - platformFee; // 개인 판매자(교사 포함)는 원천징수 대상 if (type === 'active_teacher' || type === 'individual') { const taxAmount = taxableIncome * TAX_RATE; const payout = taxableIncome - taxAmount; return { payout, taxAmount, platformFee }; } // 사업자는 세금계산서 발행 (원천징수 없음) return { payout: taxableIncome, taxAmount: 0, platformFee }; }
