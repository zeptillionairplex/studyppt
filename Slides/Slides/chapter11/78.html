export enum SellerType { ACTIVE_TEACHER = 'active_teacher', // 현직 교원 (국공립/사립) ACTIVE_PROFESSOR = 'active_professor', // 현직 교수 FORMER_TEACHER = 'former_teacher', // 전직 교원 ACADEMY_INSTRUCTOR = 'academy_instructor', // 학원 강사 GENERAL_CREATOR = 'general_creator', // 일반 제작자 INSTITUTION = 'institution' // 기관/출판사 } export interface SellerProfile { userId: string; type: SellerType; permitVerified: boolean; // 겸직허가 확인 여부 permitExpiresAt?: Date; // 허가 만료일 }
services/permissionService.ts
LOGIC
function canSellPaidContent(profile: SellerProfile): boolean { // 1. 현직 교원/교수인 경우 엄격 체크 if (profile.type === SellerType.ACTIVE_TEACHER || profile.type === SellerType.ACTIVE_PROFESSOR) { // 겸직허가 미확인 시 판매 불가 (자동 차단) if (!profile.permitVerified) return false; // 허가 기간 만료 체크 if (profile.permitExpiresAt && new Date() > profile.permitExpiresAt) { return false; } } // 2. 그 외 유형은 기본 허용 return true; }
