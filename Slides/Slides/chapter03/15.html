IMPLEMENTATION & USAGE
extractAnnotations.ts
TS
function extract(node: any, annotations: any[]) { // 1. 현재 노드에 Mark가 있는지 확인 if (node.marks) { node.marks.forEach(mark => { annotations.push({ markType: mark.type, markId: mark.attrs.id, textContent: node.text, attributes: mark.attrs }); }); } // 2. 자식 노드 재귀 순회 (Recursive) if (node.content) { node.content.forEach(child => extract(child, annotations)); } }
