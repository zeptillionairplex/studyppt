TipTap 커스텀 Mark — 구현 코드

교육적 의미(Semantics)를 부여하는 Extension 구현 패턴

TypeScript Implementation
extensions/BlankTarget.ts
TS
import { Mark, mergeAttributes } from '@tiptap/core' export const BlankTarget = Mark.create({ name: 'blankTarget', addAttributes() { return { id: { default: null }, difficulty: { default: 'medium' }, // easy, medium, hard hint: { default: '' }, } }, renderHTML({ HTMLAttributes }) { return ['span', mergeAttributes( { 'data-blank-target': '', 'class': 'blank-highlight' }, HTMLAttributes ), 0] }, })
