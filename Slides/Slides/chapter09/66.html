학생 적응형 학습 DB 스키마

개인화된 학습 경험을 지원하는 데이터 모델링

PostgreSQL Schema
student_profiles (학생 프로파일)
SQL
CREATE TABLE student_profiles ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), user_id UUID NOT NULL REFERENCES users(id), subscription_tier VARCHAR(20) NOT NULL DEFAULT 'free', -- 'free' | 'standard' | 'premium' | 'elite' -- Elo 레이팅 (전체 능력치) elo_rating FLOAT NOT NULL DEFAULT 1200.0, elo_games_count INTEGER NOT NULL DEFAULT 0, elo_k_factor FLOAT NOT NULL DEFAULT 40.0, -- 학년 및 학습 목표 grade_level VARCHAR(20) NOT NULL, learning_goal_id VARCHAR(50) DEFAULT 'school_regular', created_at TIMESTAMP DEFAULT NOW() );
Core Profile
student_knowledge_states (지식 상태)
SQL
CREATE TABLE student_knowledge_states ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), student_id UUID NOT NULL REFERENCES student_profiles(id), concept_id UUID NOT NULL REFERENCES curriculum_concepts(id), -- BKT 파라미터 (단원별 확률) p_learn FLOAT NOT NULL DEFAULT 0.1, p_transit FLOAT NOT NULL DEFAULT 0.1, p_guess FLOAT NOT NULL DEFAULT 0.25, p_slip FLOAT NOT NULL DEFAULT 0.1, mastery_level VARCHAR(20) NOT NULL DEFAULT 'not_started', last_practiced TIMESTAMP, UNIQUE(student_id, concept_id) );
BKT Model
learning_logs (학습 로그)
SQL
CREATE TABLE learning_logs ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), student_id UUID NOT NULL REFERENCES student_profiles(id), question_id UUID NOT NULL REFERENCES quiz_questions(id), is_correct BOOLEAN NOT NULL, time_spent_ms INTEGER, -- 풀이 시점의 레이팅 스냅샷 (재현성 확보) student_elo_before FLOAT NOT NULL, student_elo_after FLOAT NOT NULL, question_elo_before FLOAT NOT NULL, expected_score FLOAT NOT NULL, was_ai_generated BOOLEAN DEFAULT FALSE, created_at TIMESTAMP DEFAULT NOW() );
Analytics Data
Indexes (성능 최적화)
SQL
-- 학습 이력 조회 최적화 (시계열 분석용) CREATE INDEX idx_learning_logs_student ON learning_logs(student_id, created_at DESC); -- 특정 문제에 대한 정답률 분석용 CREATE INDEX idx_learning_logs_question ON learning_logs(question_id, is_correct); -- 단원별 지식 상태 조회 (문제 추천용) CREATE INDEX idx_knowledge_states_student ON student_knowledge_states(student_id); -- AI 생성 문제 비용 분석용 (부분 인덱스) CREATE INDEX idx_learning_logs_ai_cost ON learning_logs(created_at) WHERE was_ai_generated = TRUE;
