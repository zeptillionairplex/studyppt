-- 가장 중요한 Partial Index: is_stale=true인 행만 인덱싱하여 크기 최소화 & 속도 극대화 CREATE INDEX idx_quiz_sets_stale ON quiz_sets(is_stale) WHERE is_stale = true; -- 핵심 외래키 인덱스 (Join 성능 보장) CREATE INDEX idx_doc_versions_doc_id ON document_versions(document_id); CREATE INDEX idx_annotations_version ON document_annotations(document_version_id); -- 검수 워크플로용 상태 인덱스 CREATE INDEX idx_quiz_questions_status ON quiz_questions(status);
