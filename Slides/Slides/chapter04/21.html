핵심 DB 스키마 — PostgreSQL

Single Source of Truth와 파생물 격리를 위한 테이블 설계

PostgreSQL 15+
1. 원문 관리 (Core)
SQL
CREATE TABLE documents ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), is_archived BOOLEAN DEFAULT FALSE, -- Soft Delete delete_scheduled_at TIMESTAMPTZ -- 30일 후 삭제 ); CREATE TABLE document_versions ( id UUID PRIMARY KEY, document_id UUID REFERENCES documents(id), version_number INTEGER NOT NULL, content_json JSONB NOT NULL, -- TipTap JSON (SSOT) UNIQUE(document_id, version_number) ); CREATE TABLE document_annotations ( id UUID PRIMARY KEY, version_id UUID REFERENCES document_versions(id), mark_type VARCHAR(50), -- 'blankTarget', 'quizZone' attributes JSONB, -- {difficulty: 'easy'} text_content TEXT -- 검색용 텍스트 평탄화 );
2. 파생 콘텐츠 (Derived)
SQL
CREATE TABLE quiz_sets ( id UUID PRIMARY KEY, document_id UUID REFERENCES documents(id), source_version_id UUID REFERENCES document_versions(id), is_stale BOOLEAN DEFAULT FALSE -- 캐시 무효화 플래그 ); CREATE TABLE quiz_questions ( id UUID PRIMARY KEY, quiz_set_id UUID REFERENCES quiz_sets(id), source_annotation_id UUID, status VARCHAR(20) DEFAULT 'draft', original_text_snapshot TEXT, -- 원문 삭제 대비 안전장치 elo_rating FLOAT DEFAULT 1200.0 ); CREATE TABLE fill_blank_items ( question_id UUID REFERENCES quiz_questions(id), full_sentence TEXT, blanks JSONB -- [{"pos":0, "ans":"이산화탄소"}] );
3. 동기화 및 상태 관리 (Sync & Logs)
SQL
-- 원문 변경 영향 추적 (Cascade Control) CREATE TABLE derivation_log ( id UUID PRIMARY KEY, document_id UUID, old_version_id UUID, new_version_id UUID, affected_quiz_sets UUID[], -- 영향받은 퀴즈 세트 배열 action_taken VARCHAR(50), -- 'marked_stale', 'regenerated' created_at TIMESTAMPTZ DEFAULT NOW() );
-- AI 생성 이력 및 비용 추적 CREATE TABLE question_generation_log ( id UUID PRIMARY KEY, quiz_question_id UUID REFERENCES quiz_questions(id), model_used VARCHAR(100), -- 'gemini-flash', 'claude-sonnet' input_tokens INTEGER, output_tokens INTEGER, cost_usd DECIMAL(10,6), -- 실제 비용 ($) prompt_hash VARCHAR(64) -- 프롬프트 버전 관리 );
