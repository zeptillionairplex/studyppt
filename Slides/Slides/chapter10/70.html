DB SCHEMA: STORE CORE TABLES
store_products
SQL
CREATE TABLE store_products ( id UUID PRIMARY KEY, seller_id UUID REFERENCES users(id), title VARCHAR(200) NOT NULL, price INTEGER NOT NULL, -- 0 = 무료 category VARCHAR(50) NOT NULL, -- 핵심: 원문과 문제 세트 포함 document_ids UUID[], quiz_set_ids UUID[], status VARCHAR(20) DEFAULT 'draft', rating_avg FLOAT DEFAULT 0 );
store_purchases
SQL
CREATE TABLE store_purchases ( id UUID PRIMARY KEY, buyer_id UUID REFERENCES users(id), product_id UUID REFERENCES store_products(id), -- 수익 분배 기록 price_paid INTEGER NOT NULL, seller_revenue INTEGER NOT NULL, platform_fee INTEGER NOT NULL, purchased_at TIMESTAMP DEFAULT NOW() );
