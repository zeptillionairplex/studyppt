검수 워크플로 DB 스키마

Human-in-the-Loop 검수를 위한 데이터 모델 확장

PostgreSQL Schema Extension
schema/03_quiz_questions_alter.sql
SQL
-- quiz_questions 테이블 확장: 상태 관리 및 검수자 추적 ALTER TABLE quiz_questions ADD COLUMN status VARCHAR(20) DEFAULT 'draft' CHECK (status IN ('draft', 'in_review', 'approved', 'rejected', 'archived')); ALTER TABLE quiz_questions ADD COLUMN reviewed_by UUID REFERENCES users(id), reviewed_at TIMESTAMPTZ, rejection_reason TEXT, generation_attempt INTEGER DEFAULT 1; -- 인덱스 추가 (학생 조회용) CREATE INDEX idx_quiz_questions_approved ON quiz_questions(id) WHERE status = 'approved';
schema/04_generation_method.sql
SQL
-- AI 생성 vs 교사 수정 구분 (품질 지표 추적용) ALTER TABLE quiz_questions ADD COLUMN generation_method VARCHAR(50) DEFAULT 'auto_ai'; /* generation_method 값의 의미: - 'auto_ai': AI 생성 그대로 승인 (1차 통과) - 'auto_rule': 규칙 기반 자동 생성 - 'ai_then_manual': AI 초안 후 선생님이 수정 - 'manual': 선생님이 처음부터 직접 작성 */ -- 통계용 인덱스 CREATE INDEX idx_quiz_gen_method ON quiz_questions(generation_method);
schema/05_generation_log.sql
SQL
-- AI 생성 이력 (비용 추적 및 재생성 디버깅) CREATE TABLE question_generation_log ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), quiz_question_id UUID REFERENCES quiz_questions(id), attempt_number INTEGER NOT NULL, -- 시도 횟수 model_used VARCHAR(100) NOT NULL, -- 'claude-3-5-sonnet' 등 cost_usd DECIMAL(10,6), -- 실제 발생 비용 input_tokens INTEGER, output_tokens INTEGER, prompt_hash VARCHAR(64), -- 프롬프트 버전 raw_response JSONB, -- AI 원본 응답 additional_instruction TEXT, -- 재생성 시 피드백 created_at TIMESTAMPTZ DEFAULT NOW() );
