import git from 'isomorphic-git' import fs from 'fs' const BACKUP_DIR = '/data/git-backup' export async function backupToGit( docId: string, content: any, author: string, msg: string ) { const filepath = `documents/${docId}.json` // 1. 파일 시스템에 JSON 쓰기 (PostgreSQL 저장 후 실행) await fs.promises.writeFile( `${BACKUP_DIR}/${filepath}`, JSON.stringify(content, null, 2) ) // 2. 로컬 Git 커밋 (isomorphic-git 사용) await git.add({ fs, dir: BACKUP_DIR, filepath }) await git.commit({ fs, dir: BACKUP_DIR, message: `[${docId}] ${msg}`, author: { name: author, email: `${author}@edu.kr` } }) }
