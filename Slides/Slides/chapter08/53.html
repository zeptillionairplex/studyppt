import { create } from 'jsondiffpatch' const diffpatcher = create({ // TipTap JSON 최적화: id 기반 매칭 objectHash: (obj: any) => obj.type + (obj.attrs?.id || ''), arrays: { detectMove: true, includeValueOnMove: false }, }) export function computeDiff(oldJSON: any, newJSON: any) { return diffpatcher.diff(oldJSON, newJSON) } export function getTextChanges(oldJSON: any, newJSON: any) { const delta = computeDiff(oldJSON, newJSON) const changes = [] // delta를 재귀 순회하여 텍스트 변경(add/remove)만 추출 traverseDelta(delta, [], changes) return changes }
components/DiffViewer.tsx
TSX
import { useEditor, EditorContent } from '@tiptap/react' import { Decoration, DecorationSet } from '@tiptap/pm/view' import { Plugin } from '@tiptap/pm/state' function DiffViewer({ oldVersion, newVersion }) { const changes = getTextChanges(oldVersion.content, newVersion.content) const editor = useEditor({ content: newVersion.content, editable: false, extensions: [ Extension.create({ addProseMirrorPlugins() { return [new Plugin({ props: { decorations(state) { const decos = changes.map(c => { if (c.type === 'added') return Decoration.inline(c.from, c.to, { class: 'diff-added' }) if (c.type === 'removed') return Decoration.inline(c.from, c.to, { class: 'diff-removed' }) }) return DecorationSet.create(state.doc, decos.filter(Boolean)) } } })] } }) ] }) return <EditorContent editor={editor} /> }

Logic Overview
NOTE
Implementation Strategy: 1. jsondiffpatch calculates structural diff between two JSONs. - Use objectHash to correctly identify paragraphs/nodes even if moved. 2. getTextChanges() traverses the delta to flatten complex JSON diffs into simple text ranges: - { type: 'added', from: 10, to: 15 } - { type: 'removed', from: 22, to: 28 } 3. ProseMirror Decoration API applies CSS classes to these ranges without changing the document content. // Result: Users see "track changes" style visualization directly in the editor.
